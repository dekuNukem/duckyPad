DEFAULTCHARDELAY 25
DEFAULTDELAY 25

DEFINE PLUS_BUTTON 17

FUNCTION blink()
    VAR $count = 0
    WHILE $count < 4
        SWC_FILL 100 100 100
        DELAY 200
        SWC_FILL 0 0 0
        DELAY 200
        $count = $count + 1
    END_WHILE
END_FUNCTION

VAR $THISKEY = 0
FUNCTION BLOCKING_READKEY()
    $THISKEY = 0
    BCLR
    WHILE 1
        DELAY 50
        $THISKEY = $_READKEY
        IF $THISKEY != 0 THEN
            LBREAK
        END_IF
    END_WHILE
END_FUNCTION

blink()
