WHILE 1
	OLED_CLEAR
	OLED_CURSOR 20 0
	OLED_PRINT Hello World!
	OLED_CURSOR 20 12
	OLED_PRINT Press a key!
	OLED_CURSOR 10 52
	OLED_PRINT Press + to exit

	VAR $this_key = $_READKEY

	IF $this_key != 0 THEN
		OLED_CURSOR 10 31
		OLED_PRINT I pressed key $this_key!
	END_IF

	IF $this_key == 17 THEN
		BREAK
	END_IF

	OLED_UPDATE
	DELAY 50
END_WHILE
